Чекер

Сигнатура метода в интерфейсе IChecker выглядит следующим образом:

void Check(List<string> modeOutput, string pathToUserProgram, string pathToModelProgram, ref ResultRun result);

Параметр modelOutput – это список строк, хранящий стандартный вывод эталонного приложения.

Параметр pathToUserProgram – это строка, хранящая абсолютный путь к исполняемому файлу пользователя.

Параметр pathToModelProgram – это строка, хранящая абсолютный путь к исполняемому файлу эталонной программы.

Параметр по ссылке result – это структура, представляющая из себя результаты одного тестового прогона. Эта структура содержит множество полей:

·         long Memory – количество потребляемой пользователем памяти в байтах в текущим тестовом прогоне;

·         int Time – время работы пользовательской программы в миллисекундах текущим тестовом прогоне;

·         bool IsCorrect – параметр представляющий собой правильность работы пользовательской программы при заданных входных данных (преподаватель должен обязательно изменить этот параметр, если параметр не изменён тестовый проход считается ошибочным т.е. аргумент равен false);

·         string Comment – комментарий к тестовому прогону, который будет выведен пользователю прим. “”;

·         List<string> Input – Список строк, представляющие входные данные из стандартного потока ввода пользовательской программы;

·         List<string> Output – Список строк, представляющий выходные данные из стандартного потока вывода пользовательской программы;

Пример написания чекера

// Подключим нужные компоненты для чекера

using System.Collections.Generic;

using System.IO;

using ALS.CheckModule.Compare.Checker;

using ALS.CheckModule.Compare.DataStructures;

 

// Укажем пространство имён этой компоненты

namespace ALS.CheckModule.Component.Checker

{

   // Создадим класс и реализуем интерфейс IChecker

     public class AbsoluteChecker: IChecker

    {

        // Реализуемый метод из интерфейса

        public void Check (List<string> modeOutput,

    string pathToUserProgram,

    string pathToModelProgram,

    ref ResultRun result)

        {

            // Заведём счётчик для того, чтобы узнать на какой позиции пользовательского  

           // вывода проблемы

            var counter = 0;

            // Используем совместный цикл для прохождения по множеству     

            // пользовательского вывода

            foreach (var userElem in result.Output)

            {

            // Если элемент стандартного вывода пользователя не равен эталонному элементу                 

    if (userElem != modeOutput[counter])

                {

                    // Запишем комментарий для пользователя, который будет выведен ему и 

                   // прекратим работу чекера отдав управление системе

                    result.Comment = $"Ошибка на позиции {counter}:
                                                     ожидалось{ modeOutput [counter] }, получено {userElem}";

                    return;

                }

                ++counter;

            }

           

            // Если пользовательский вывод содержал меньше значений, чем требуется

            if (counter < modeOutput.Count)

            {

                //Запишем комментарий об этом и так же прекратим работу чекера

                result.Comment = $"Количество выводов в программе {counter}

ожидалось {modeOutput.Count}";

                return;

            }

            //Получим потоки чтения для пользовательской программы и эталонной

            using var srUser = new StreamReader(GetFilePath(pathToUserProgram));

            using var srModel = new StreamReader(GetFilePath(pathToModelProgram));

 

 

            //Сравним пользовательский вывод в файл и эталонный

            if (srUser.ReadToEnd() != srModel.ReadToEnd())

            {

                result.Comment = $"Выводы в файлах не совпадают";

                return;

            }

           // Если все проверки до этого были успешны запишем комментарий об успешности

           // выполненного теста и пометим флаг IsCorret в значение true, если этого не 

           // сделать тестовый прогон будет считаться ошибочным и в следствии вся задача

           // будет   оценена в 0 из N, где N – количество тестовых наборов

            result.Comment = "Всё хоккей";

            result.IsCorrect = true;

        }

        //Функция для получения выходного файла

        private string GetFilePath(string pathToProgram)

        {

            return Path.Combine(Directory.GetParent(pathToProgram).ToString(), "output.txt");

        }

    }

}
